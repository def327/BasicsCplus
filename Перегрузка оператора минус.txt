#include <iostream>

using namespace std;

  class coord
  {
      int x;
      int y;

  public:
       coord()
          {
            x=0;
            y=0;
          }
        coord(int xm,int ym)
          {
              x=xm;
              y=ym;
          }
        void  show()
            {
                cout<<"x="<<x<<"    "<<"y="<<y;
            }
         coord operator-(const coord &obj);
  };
int main()
{
    coord ob1(4,4);
    coord ob2(4,4);
    coord ob3;
    ob3=ob1-ob2-ob1;
    ob3.show();

}

coord coord::operator-(const coord &obj)
    {
        coord tmp;          //создаём буфер хранения
        tmp.x=x-obj.x;
        tmp.y=y-obj.y;
        return tmp;
    }
