#include <iostream>
#include <vector>
using namespace std;

int main()
{
    vector <int> v; //шаблон типа int
    cout<<v.size()<<endl;
    for(int i=0;i<10;i++)
      {
        v.push_back(i);
      }
    cout<<v.size();
    for(int i=0;i<10;i++)
      {
        cout<<v[i]<<endl;
      }
 vector<int>::iterator p=v.begin(); //объявление указателя на начало вектора
      while(p!=v.end()) //цикл пока значения вектора не достанет конца
        {
          cout<<*p;
          p++;
        }
 vector<int>::iterator t=v.begin(); //объявление указателя на начало вектора
        {
          t+=2;
          v.insert(t,10,9);       //вставить после 2-го элемента десять 9ок подряд
        }
        cout<<endl;

 vector<int>::iterator y=v.begin();
         {
          y+=2;
          v.erase(y,y+10);       //вставить после 2-го элемента десять 9ок подряд
         }
         cout<<endl;
 vector<int>::iterator x=v.begin();
      while(x!=v.end()) //цикл пока значения вектора не достанет конца
        {
          cout<<*x;
          x++;
        }
}
